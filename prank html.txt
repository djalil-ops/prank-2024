  <script>
    const images = ["img1", "img2", "img3"];
    const container = document.getElementById('container');
    const audio = document.getElementById('audio');
    const glitch = document.querySelector('.glitch');
    const flash = document.querySelector('.flash');

    let currentIndex = 0;

    function triggerGlitchEffect() {
      glitch.style.opacity = '1';
      setTimeout(() => {
        glitch.style.opacity = '0';
      }, 200);
    }

    function triggerFlashEffect() {
      flash.style.opacity = '1';
      setTimeout(() => {
        flash.style.opacity = '0';
      }, 100);
    }

    function showNextImage() {
      if (currentIndex > 0) {
        const prevImage = document.getElementById(images[currentIndex - 1]);
        prevImage.style.display = 'none';
        prevImage.style.transform = 'scale(1) rotate(0deg)';
      }

      if (currentIndex < images.length) {
        const currentImage = document.getElementById(images[currentIndex]);
        currentImage.style.display = 'block';
        currentImage.style.transform = `scale(${1 + Math.random() * 0.2}) rotate(${Math.random() * 20 - 10}deg)`;
        triggerGlitchEffect();
        triggerFlashEffect();
        setTimeout(showNextImage, 2000);
        currentIndex++;
      } else {
        container.style.display = 'none';
        audio.pause();
      }
    }

    function startPrank() {
      audio.play();
      container.style.display = 'flex';
      showNextImage();
    }

    // Start prank after a slight delay
    setTimeout(startPrank, 3000);
  </script>
